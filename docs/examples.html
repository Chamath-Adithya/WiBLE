Examples

Learn by doing. Copy-paste ready code.

1. Basic Provisioning

The simplest way to get started.

BasicProvisioning.ino 
#include <WiBLE.h>

WiBLE provisioner;

void setup() {
    Serial.begin(115200);
    ProvisioningConfig config;
    config.securityLevel = SecurityLevel::SECURE;
    provisioner.begin(config);
    
    provisioner.onWiFiConnected([](String ssid, String ip) {
        Serial.printf("Connected to %s (IP: %s)\n", ssid.c_str(), ip.c_str());
    });
    
    provisioner.startProvisioning();
}

void loop() {
    provisioner.loop();
}
