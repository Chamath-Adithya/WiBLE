Interactive Configurator

main.cpp 
#include <WiBLE.h>

WiBLE provisioner;

void setup() {
    Serial.begin(115200);
    
    ProvisioningConfig config;
    config.deviceName = "My-ESP32";
    config.securityLevel = SecurityLevel::SECURE;
    
    provisioner.begin(config);
    provisioner.startProvisioning();
}

void loop() {
    provisioner.loop();
}
